# 1. Go to your project folder
cd path/to/your/project

# 2. Create a virtual environment named 'venv'
python3 -m venv venv (or) python venv venv  --> for window

# 3. Activate the virtual environment (macOS/Linux)
source venv/bin/activate

# 4. Activate the virtual environment (Windows CMD)
venv\Scripts\activate

# 5. Activate the virtual environment (Windows PowerShell)
.\venv\Scripts\Activate.ps1

# 6. Install required packages (if you have requirements.txt)
pip install -r requirements.txt

# 7. Or install packages manually (example: flask and requests)
pip install flask requests

# 8. Run your Python application
python app.py

# 9. Save installed packages to requirements.txt
pip freeze > requirements.txt

# 10. Deactivate the virtual environment when done
deactivate

| Step | Command                              | Purpose                                              |
| ---- | ------------------------------------ | ---------------------------------------------------- |
| 1    | `cd my_project`                      | Go to project folder                                 |
| 2    | `uv venv`                            | Create virtual environment with default name `.venv` |
| 3    | `uv venv myenv`                      | Create virtual environment with custom name          |
| 4    | `source .venv/bin/activate`          | Activate venv in Linux/macOS                         |
| 5    | `.venv\Scripts\activate.bat`         | Activate venv in Windows CMD                         |
| 6    | `.venv\Scripts\Activate.ps1`         | Activate venv in Windows PowerShell                  |
| 7    | `uv pip install selenium`            | Install package using UV                             |
| 8    | `uv add selenium`                    | Add dependency to project                            |
| 9    | `uv pip freeze > requirements.txt`   | Create `requirements.txt` file                       |
| 10   | `uv pip install -r requirements.txt` | Install packages from requirements file              |
| 11   | `rm -rf .venv`                       | Remove virtual environment in Linux/macOS            |
| 12   | `rmdir /s .venv`                     | Remove virtual environment in Windows                |
